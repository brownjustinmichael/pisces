163  virtual void initialize (int name, double* initial_conditions = NULL) {
164  TRACE ("Initializing " << name);
165  if (name == position && !initial_conditions) {
166  double pioN = std::acos (-1.0) / (n - 1);
167  double scale = (position_0 - position_n) / 2.0;
168  double initial_position = (position_0 + position_n) / 2.0;
169  std::vector <double> init (n);
170  for (int i = 0; i < n; ++i) {
171  init [i] = scale * std::cos (i * pioN) + initial_position;
172  }
173  one_d::element::initialize (name, &init [0]);
174  } else if (name == velocity && !initial_conditions){
175  double scale = inputParams["init_cond_scale"].asDouble;
176  double width = inputParams["init_cond_width"].asDouble;
177  double mean = inputParams["init_cond_mean"].asDouble;
178  double sigma = inputParams["init_cond_sigma"].asDouble;
179  std::vector <double> init (n);
180  double height, temp;
181  height = std::max (scale * std::exp (- (width / 2.0 - mean) * (width / 2.0 - mean) / 2.0 / sigma / sigma), scale * std::exp (- (- width / 2.0 - mean) * (- width / 2.0 - mean) / 2.0 / sigma / sigma));
182  for (int i = 0; i < n; ++i) {
183  temp = scale * std::exp (- ((*this) (position, i) - mean) * ((*this) (position, i) - mean) / 2.0 / sigma / sigma) - height;
184  if (temp > 0.0) {
185  init [i] = temp;
186  } else {
187  init [i] = 0.0;
188  }
189  }
190  one_d::element::initialize (name, &init [0]);
191  } else {
192  one_d::element::initialize (name, initial_conditions);
193  }
194  TRACE ("Initialized.");
195  }
