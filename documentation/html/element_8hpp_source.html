236  virtual void solve () {
237  double t_timestep;
238  t_timestep = calculate_timestep ();
239  messenger_ptr->min (&t_timestep);
240 
241  for (std::shared_ptr <solver> i_solver : solvers) {
242  i_solver->execute ();
243  }
244 
245  duration += timestep;
246  INFO ("TOTAL TIME: " << duration);
247  if (t_timestep != timestep) {
248  flags &= ~unchanged_timestep;
249  INFO ("Updating timestep: " << t_timestep);
250  } else {
251  flags |= unchanged_timestep;
252  }
253  timestep = t_timestep;
254  flags |= transformed;
255  }
